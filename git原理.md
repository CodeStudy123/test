# 本地库保存版本的机制：
# 集中式svn： A1A2A3(文件的修改部分的内容)
rootversion1   version2   version3   version4   version5
  fileA ------------------------------------->A1----------->A2     
  fileB -------->B1-------------------------->B2--------------
  fileC -------->C1-------------------------->C2----------->C3
  svn每次提交本地库都会生成一条记录（版本），每个版本中只会保存当前版本相比之前版本修改的部分
  所以当要访问某个版本时：它会整合之前所有版本，拼到一块儿

# 分布式git： A1A2A3(修改后的文件)
rootversion1   version2   version3   version4   version5
  fileA      A1  <--  A1      A2   <--  A2
  fileB  <--   B     B1  <--   B1       B2
  fileC  <--   C      C1      C2      C3
  git每次提交本地库都会生成一条记录（版本），每个版本都会保存全部文件的快照
  如果文件与之前版本没有区别，则创建指针指向之前的版本（节约内存空间）

# 分支：
  新建分支：就是新建指针，指向某个版本
  切换分支：改变HEAD的指向，指向新分支
